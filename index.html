javascript oop guide

basic object oriented programming

1. this keyword

the 'this' keyword points to different objects depending on how you call the function.

rules:

  1. method call: obj.method() → this = obj
  2. function call: function() → this = window
  3. constructor: new Function() → this = new object
  4. arrow function: this = outer scope
// examples const obj = { value: 42, regular: function() { return this.value; // this = obj }, arrow: () => { return this.value; // this = window } }; console.log(obj.regular()); // 42 console.log(obj.arrow()); // undefined function Person(name) { this.name = name; } const p = new Person('alice'); console.log(p.name); // alice function greet(msg) { return msg + ' ' + this.name; } const user = { name: 'bob' }; greet.call(user, 'hi'); // hi bob greet.apply(user, ['hey']); // hey bob const bound = greet.bind(user); bound('hello'); // hello bob

2. making objects

four ways to create objects in javascript.

object literal

const player = { name: 'hero', hp: 100, attack: function() { return this.name + ' attacks'; } };

constructor function

function Enemy(name, hp) { this.name = name; this.hp = hp; } Enemy.prototype.attack = function() { return this.name + ' attacks'; }; const orc = new Enemy('orc', 50);

object.create

const proto = { attack: function() { return this.name + ' attacks'; } }; const goblin = Object.create(proto); goblin.name = 'goblin'; goblin.hp = 30;

es6 class

class Monster { constructor(name, hp) { this.name = name; this.hp = hp; } attack() { return this.name + ' attacks'; } } const troll = new Monster('troll', 80);

3. inheritance

sharing properties and methods between objects.

prototype chain

function Animal(name) { this.name = name; } Animal.prototype.speak = function() { return this.name + ' makes sound'; }; function Dog(name) { Animal.call(this, name); } Dog.prototype = Object.create(Animal.prototype); Dog.prototype.constructor = Dog; Dog.prototype.speak = function() { return this.name + ' barks'; }; const dog = new Dog('rex');

es6 class inheritance

class Vehicle { constructor(brand) { this.brand = brand; } start() { return this.brand + ' started'; } } class Car extends Vehicle { constructor(brand, doors) { super(brand); this.doors = doors; } start() { return this.brand + ' car started'; } } const car = new Car('toyota', 4);

4. comet assignment

make comet objects for space game.

version 1: object.create

const cometProto = { rnd: (max) => Math.round(Math.random() * max), move: function() { this.x += this.vx; this.y += this.vy; if (this.x < 0) this.x = 400; if (this.x > 400) this.x = 0; if (this.y < 0) this.y = 300; if (this.y > 300) this.y = 0; } }; function makeComet() { const c = Object.create(cometProto); c.size = cometProto.rnd(40) + 10; c.x = cometProto.rnd(350); c.y = cometProto.rnd(250); c.vx = cometProto.rnd(4) - 2; c.vy = cometProto.rnd(4) - 2; return c; }

version 2: constructor

function Comet() { this.size = Math.round(Math.random() * 40) + 10; this.x = Math.round(Math.random() * 350); this.y = Math.round(Math.random() * 250); this.vx = Math.round(Math.random() * 4) - 2; this.vy = Math.round(Math.random() * 4) - 2; } Comet.prototype.move = function() { this.x += this.vx; this.y += this.vy; if (this.x < 0) this.x = 400; if (this.x > 400) this.x = 0; if (this.y < 0) this.y = 300; if (this.y > 300) this.y = 0; };

version 3: es6 class

class SpaceComet { constructor() { this.size = Math.round(Math.random() * 40) + 10; this.x = Math.round(Math.random() * 350); this.y = Math.round(Math.random() * 250); this.vx = Math.round(Math.random() * 4) - 2; this.vy = Math.round(Math.random() * 4) - 2; } move() { this.x += this.vx; this.y += this.vy; if (this.x < 0) this.x = 400; if (this.x > 400) this.x = 0; if (this.y < 0) this.y = 300; if (this.y > 300) this.y = 0; } draw(ctx) { ctx.fillStyle = 'white'; ctx.fillRect(this.x, this.y, this.size, this.size); } }

5. ball class

svg ball objects with click events.

class Ball { constructor(size, x, y) { this.size = size; this.x = x; this.y = y; this.color = this.randomColor(); this.clicks = 0; } randomColor() { const colors = ['red', 'blue', 'green', 'orange']; return colors[Math.floor(Math.random() * colors.length)]; } draw(svg) { const circle = document.createElementNS("http://www.w3.org/2000/svg", "circle"); circle.setAttribute("cx", this.x); circle.setAttribute("cy", this.y); circle.setAttribute("r", this.size); circle.setAttribute("fill", this.color); circle.setAttribute("stroke", "black"); circle.style.cursor = 'pointer'; circle.onclick = () => { this.clicks++; this.color = this.randomColor(); circle.setAttribute("fill", this.color); }; svg.appendChild(circle); } }

6. advanced

prototype chain: javascript looks for properties in object → prototype → parent prototype → object.prototype

call, apply, bind

function attack(weapon, damage) { return this.name + ' uses ' + weapon + ' for ' + damage; } const hero = { name: 'knight' }; attack.call(hero, 'sword', 20); attack.apply(hero, ['bow', 15]); const heroAttack = attack.bind(hero); heroAttack('axe', 25);

prototype chain

function A() {} A.prototype.methodA = function() { return 'from A'; }; function B() {} B.prototype = Object.create(A.prototype); B.prototype.methodB = function() { return 'from B'; }; const obj = new B(); obj.methodA(); // from A obj.methodB(); // from B
